▸ TEKNOFEST API ENTEGRASYONU

Şartname gereği HttpClient ile JSON POST. Telemetri gönderim sıklığı en fazla 2 Hz (2 Hz üstü 400 hatası).

public class TeknofestApi
{
    private readonly HttpClient _http;
    private string _cookie;

    public async Task Giris(string kullaniciAdi, string sifre)
    {
        var body = new { kadi = kullaniciAdi, sifre };
        var res = await _http.PostAsJsonAsync("/api/giris", body);
        _cookie = res.Headers.GetValues("Set-Cookie").FirstOrDefault();
    }

    // 1 Hz timer ile çağrılır (2 Hz'den fazla OLMASIN)
    public async Task<TelemetriCevap> TelemetriGonder(TelemetriIstek t)
    {
        var res = await _http.PostAsJsonAsync("/api/telemetri_gonder", t);
        return await res.Content.ReadFromJsonAsync<TelemetriCevap>();
    }

    public async Task<List<HssKoordinat>> HssKoordinatlari()
        => await _http.GetFromJsonAsync<List<HssKoordinat>>("/api/hss_koordinatlari");

    public async Task KilitlenmeBildir(KilitlenmeBilgisi k)
        => await _http.PostAsJsonAsync("/api/kilitlenme_bilgisi", k);

    public async Task KamikazeBildir(KamikazeBilgisi k)
        => await _http.PostAsJsonAsync("/api/kamikaze_bilgisi", k);
}

// Telemetri istek modeli (şartname Ek-1)
public class TelemetriIstek
{
    public int takim_numarasi { get; set; }
    public double iha_enlem { get; set; }
    public double iha_boylam { get; set; }
    public double iha_irtifa { get; set; }

    public int hedef_merkez_X { get; set; }
    public int hedef_merkez_Y { get; set; }
    public int hedef_genislik { get; set; }

▸ İHA İLE HABERLEŞME - MAVLink

Pixhawk/ArduPilot otopilotu ile konuşmak için MAVLink.NET veya MAVSDK. UDP üzerinden telemetri alınır.

<PackageReference Include="MAVLink" Version="1.0.12" />
using MAVLink;

var parser = new MavlinkParse();
var udp = new UdpClient(14550);

while (true)
{
    var data = await udp.ReceiveAsync();
    var packet = parser.ReadPacket(new MemoryStream(data.Buffer));

    if (packet?.data is mavlink_global_position_int_t pos)
    {
        OwnAircraft.Lat = pos.lat / 1e7;
        OwnAircraft.Lng = pos.lon / 1e7;
        OwnAircraft.Alt = pos.relative_alt / 1000.0;
    }
    else if (packet?.data is mavlink_attitude_t att)
    {
        Flight.Roll = att.roll * 180.0 / Math.PI;
        Flight.Pitch = att.pitch * 180.0 / Math.PI;
    }
}

▸ İPUÇLARI

  • Performans: Haritada 50+ marker olunca VirtualizingPanel.IsVirtualizing="True" kullan.
  • Thread safety: MAVLink UDP loop'u arka thread'de; UI güncellemesi için Dispatcher.Invoke.
  • Telemetri timer'ı: DispatcherTimer 1000ms. Asla 2 Hz üstüne çıkma - şartname 400 hatası veriyor.
  • HSS durumu: /api/hss_koordinatlari boş liste dönebilir - hakem duyurmadan önce. Null check şart.
  • Kilitlenme: Hedef kutusu ekranda minimum hedef_genislik × hedef_yukseklik olmalı ve hedef_merkez_X/Y ekran merkezine yeterince yakın olmalı (şartnameye bakın).
  • GPS saati: gps_saati formatı zorunlu — DateTime.UtcNow.ToString("HH:mm:ss.fff").
  • Cookie yönetimi: HttpClientHandler.CookieContainer ile otomatik. Manuel header atma.
